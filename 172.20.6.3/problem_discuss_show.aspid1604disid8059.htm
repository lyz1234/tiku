这道题慎重使用Tarjan
周航2013
由于数据量，记忆化搜索或者倒序递推都可以不超时完成，但是在做Tarjan的时候会系统堆栈溢出，所以只能过9组

【代码如下】
/*
Author : Zhou Hang
For : Tarjan
Data : 2012-4-13
*/

#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MaxN=100100;

int totedge=0;
struct edge
{
int y,next;
} e[MaxN];
int first[MaxN],out[MaxN];
int list[MaxN],open,closd;

int n;
int totcorlor,top,index;
int numcorlor[MaxN],ans[MaxN];
int next[MaxN],corlor[MaxN];
int stack[MaxN];
int dfn[MaxN],low[MaxN];
bool instack[MaxN],v[MaxN];
//保证每次处理一种颜色的时候每个色块只访问一次
bool onlyone[MaxN];

void init()
{
memset(dfn,0,sizeof(dfn));
memset(low,127,sizeof(low));
memset(stack,0,sizeof(stack));
memset(numcorlor,0,sizeof(numcorlor));
memset(instack,false,sizeof(instack));
index=top=totcorlor=0;
}

void Add_edge(int x,int y)
{
e[++totedge].y=y;
e[totedge].next=first[x];
first[x]=totedge;
//下语句用于倒序，为正序时的出度
out[y]++;
}

void Tarjan(int x)
{
dfn[x]=++index;
low[x]=dfn[x]; //初始low数组

if (next[x]==x)
{
totcorlor++;
corlor[x]=totcorlor;
numcorlor[totcorlor]++;
return;
}

//x点进栈
stack[++top]=x;
instack[x]=true;

int tn=next[x];
if (dfn[tn]==0)
{
Tarjan(tn);
low[x]=min(low[x],low[tn]);
}
else if ((dfn[tn]!=0)&&(instack[tn]))
low[x]=min(dfn[tn],low[x]);

if (dfn[x]==low[x])
{
int tmp;
totcorlor++;
while (stack[top+1]!=x)
{
tmp=stack[top--];
corlor[tmp]=totcorlor;
instack[tmp]=false;
numcorlor[totcorlor]++;
}
}
}


// k=1用于递归的建图 k=2倒序递推的建图
void Before_Work(int k)
{
int nowcor,nextcor;
for (int i=1; i<=n; i++)
{
nowcor=corlor[i];
nextcor=corlor[next[i]];
if (nowcor!=nextcor)
if (k==1) Add_edge(nowcor,nextcor);
else Add_edge(nextcor,nowcor);
}
}

void Search(int x)
{
if (v[x]) return;
v[x]=true;

for (int te=first[x]; e[te].y!=0; te=e[te].next)
{
int tn=e[te].y;
Search(tn);
numcorlor[x]+=numcorlor[tn];
}
}
*/

//用于递推
closd=open=0;
for (int i=1; i<=totcorlor; i++)
if (out[i]==0) list[++open]=i;
while (closd<open)
{
int x=list[++closd];
memset(onlyone,false,sizeof(onlyone));
for (int te=first[x]; e[te].y!=0; te=e[te].next)
{
int tn=e[te].y;
if (!onlyone[tn])
