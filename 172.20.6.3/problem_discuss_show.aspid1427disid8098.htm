讨论 Discussion
 
【noip -- mayan】搜索
周航2013
纯搜索。

1，每次判断每种颜色，如果存在一种颜色的个数等于1或者等于2，那么就不可能消除
2，如果能左移能右移，由于在字典序上有要求，那么只进行右移，除非左边是空的。
3，如果两个颜色相同，那么搜索的时候可以跳过，左右移对答案没有影响。


【代码】（内附注释）


/**
*Prob : NOIp 2012 -- mayan
*Data : 2012-11-4
*Sol : 模拟
*Status : Accept
*/

#include <cstdio>
#include <cstring>

#define MaxN 20

using namespace std;

//棋盘形状  a[i][j]表示第i列第J个
//J由上到下 1--7
struct rec {
int a[MaxN][MaxN];
};
//记录每种颜色个数
int color[MaxN];
//记录答案
int ans[MaxN][5];

int n; rec a;

//进行消除处理
rec Clear_it(rec g)
{
//步骤1 -- 处理掉落问题
for (int i=1; i<=5; i++)
for (int j=1; j<=7; j++)
if (g.a[i][j]==0) {
for (int k=j+1; k<=7; k++)
if (g.a[i][k]!=0) {
g.a[i][j] = g.a[i][k];
g.a[i][k] = 0;
break;
}
}

//步骤2 -- 消除同色
rec tmp = g;
bool flag = false; //是否存在同色
for (int i=1; i<=5; i++)
for (int j=1; j<=7; j++)
if (g.a[i][j]!=0) {
//纵向相同
int now = 1;
for (int k=j+1; k<=7; k++)
if (g.a[i][k]==g.a[i][k-1]) now++;
else break;

if (now>2) {
tmp.a[i][j] = 0; flag = 1;
for (int k=j+1; k<=7; k++)
if (g.a[i][k]==g.a[i][k-1]) tmp.a[i][k]=0;
else break;
}
//横向相同
now = 1;
for (int k=i+1; k<=5; k++) {
if (g.a[k][j]==g.a[k-1][j]) now++;
else break;
}
if (now>2) {
tmp.a[i][j] = 0; flag = 1;
for (int k=i+1; k<=5; k++)
if (g.a[k][j]==g.a[k-1][j]) tmp.a[k][j]=0;
else break;
}
}
//重复，接着消除
if (flag) tmp = Clear_it(tmp);
//消除完毕 退出
if (flag) return tmp;
}


bool dfs(int t, rec g)
{
rec tmp;
//进行消除处理
g = Clear_it(g);

//判断是否全部消除
bool flag = true;
for (int i=1; i<=5; i++)
for (int j=1; j<=7; j++)
if (g.a[i][j]>0) flag = 0;

if (flag) return 1;
if (t>n) return 0;

//存在颜色少于3个的退出
memset(color,0,sizeof(color));
for (int i=1; i<=5; i++)
for (int j=1; j<=7; j++)
color[g.a[i][j]]++;
for (int i=1; i<=10; i++)
if (color[i]==1||color[i]==2) return 0;

//只会存在向右移动
for (int i=1; i<=4; i++)
for (int j=1; j<=7; j++)
//两个不相等且都不为空
if ( (g.a[i][j]!=0||g.a[i+1][j]!=0)&&(g.a[i][j]!=g.a[i+1][j]) ) {
tmp = g;
//交换两个方块
int temp = g.a[i][j];
tmp.a[i][j] = tmp.a[i+1][j];
tmp.a[i+1][j] = temp;
//可以消除到最后
if ( dfs(t+1,tmp) ) {
ans[t][1]=i; ans[t][2]=j; ans[t][3]=1;
//特殊情况：左边为空则只能左移
if (!g.a[i][j]) {
ans[t][1] = i+1;
ans[t][3] = -1;
}
return 1;
}
}
return 0;
}

int main()
{
freopen("mayan.in","r",stdin);
freopen("mayan.out","w",stdout);

　　scanf("%d",&n);

for (int i=1; i<=5; i++)
for (int j=1; j<=8; j++) {
scanf("%d",&a.a[i][j]);
if (a.a[i][j]==0) break;
}

　　if (dfs(1,a))
for (int i=1;i<=n;i++)
printf("%d %d %d\n",ans[i][1]-1,ans[i][2]-1,ans[i][3]);
else printf("-1\n");//输出方案即可


fclose(stdin); fclose(stdout);
return 0;
}
